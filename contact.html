const TO_EMAIL = 'teachgrenfell@soton.ac.uk'; const SHEET_NAME = 'Enquiries'; function doPost(e) { try { const params = e.parameter || {}; // Honeypot field: real users should not fill this. if (params.website) { return textResponse('Thank you. Your enquiry has been sent.'); } // Verify Cloudflare Turnstile. const turnstileToken = params['cf-turnstile-response']; if (!turnstileToken) { return textResponse('Captcha token missing. Please go back and try again.'); } if (!verifyTurnstile(turnstileToken)) { return textResponse('Captcha verification failed. Please go back and try again.'); } const name = sanitise(params.name, 200); const email = sanitise(params.email, 300); const enquiryArea = sanitise(params.enquiry_area, 200); const message = sanitise(params.message, 5000); if (!name || !email || !message) { return textResponse('Please complete all required fields.'); } if (!isValidEmail(email)) { return textResponse('Please enter a valid email address.'); } // Save to Google Sheet. const sheet = getOrCreateSheet(); sheet.appendRow([ new Date(), name, email, enquiryArea, message ]); // Send email. MailApp.sendEmail({ to: TO_EMAIL, subject: 'Grenfell Curriculum enquiry', replyTo: email, name: 'Grenfell Curriculum website', body: 'New enquiry from the Grenfell Curriculum website\n\n' + 'Name: ' + name + '\n' + 'Email: ' + email + '\n' + 'Enquiry area: ' + enquiryArea + '\n\n' + 'Message:\n' + message }); return textResponse('Thank you. Your enquiry has been sent.'); } catch (error) { console.error('FORM ERROR:', error); return textResponse( 'There was a problem sending your enquiry. Please try again later.' ); } } function verifyTurnstile(token) { const secret = PropertiesService .getScriptProperties() .getProperty('TURNSTILE_SECRET_KEY'); if (!secret) { console.error('Missing TURNSTILE_SECRET_KEY in Script Properties.'); return false; } const response = UrlFetchApp.fetch( 'https://challenges.cloudflare.com/turnstile/v0/siteverify', { method: 'post', payload: { secret: secret, response: token }, muteHttpExceptions: true } ); const resultText = response.getContentText(); console.log('Turnstile response:', resultText); const result = JSON.parse(resultText); return result.success === true; } function getOrCreateSheet() { const spreadsheetId = PropertiesService .getScriptProperties() .getProperty('SPREADSHEET_ID'); if (!spreadsheetId) { throw new Error('Missing SPREADSHEET_ID in Script Properties.'); } const spreadsheet = SpreadsheetApp.openById(spreadsheetId); let sheet = spreadsheet.getSheetByName(SHEET_NAME); if (!sheet) { sheet = spreadsheet.insertSheet(SHEET_NAME); sheet.appendRow([ 'Timestamp', 'Name', 'Email', 'Enquiry area', 'Message' ]); } return sheet; } function sanitise(value, maxLength) { return String(value || '') .replace(/[<>]/g, '') .trim() .slice(0, maxLength); } function isValidEmail(email) { return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email); } function textResponse(message) { return ContentService .createTextOutput(message) .setMimeType(ContentService.MimeType.TEXT); } // Optional: run this once manually to trigger Google authorisation. function authorise() { getOrCreateSheet(); MailApp.getRemainingDailyQuota(); UrlFetchApp.fetch('https://challenges.cloudflare.com/'); }
